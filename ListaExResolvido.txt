1 - Faça um programa para ler dois valores inteiros, e depois mostrar na tela a soma desses números com uma 
mensagem explicativa, conforme exemplos

using System;
using System.Globalization;

namespace FirtsProject {
    class Program {
        static void Main(string[] args) {


            int firstNumber = int.Parse(Console.ReadLine());
            int twoNumber = int.Parse(Console.ReadLine());

            int soma = (firstNumber + twoNumber);

            Console.Write("Soma: " + soma);
        }
    }
}


2 - Faça um programa para ler o valor do raio de um círculo, e depois mostrar o valor da área deste círculo com quatro 
casas decimais conforme exemplos.
Fórmula da área: area = π . raio2
Considere o valor de π = 3.14159

using System;
using System.Globalization;

namespace FirtsProject {
    class Program {
        static void Main(string[] args) {


            double valorRaio = double.Parse(Console.ReadLine(), CultureInfo.InvariantCulture);

            double area = 3.14159 * (Math.Pow(valorRaio, 2));

            Console.WriteLine("A=" + area.ToString("F4"), CultureInfo.CurrentCulture);

        }
    }
}

3 - Fazer um programa para ler quatro valores inteiros A, B, C e D. A seguir, calcule e mostre a diferença do produto 
de A e B pelo produto de C e D segundo a fórmula: DIFERENCA = (A * B - C * D)

using System;
using System.Globalization;

namespace FirtsProject {
    class Program {
        static void Main(string[] args) {

        }

}

4 - Fazer um programa que leia o número de um funcionário, seu número de horas trabalhadas, o valor que recebe por 
hora e calcula o salário desse funcionário. A seguir, mostre o número e o salário do funcionário, com duas casas 
decimais.

using System;
using System.Globalization;

namespace FirtsProject {
    class Program {
        static void Main(string[] args) {

            int num = 0;
            double horasTrabalhadas = 0.0;
            double valorHora = 0.0;
            double salario = 0.0;

            Console.WriteLine("Informe seu número: ");
            num = int.Parse(Console.ReadLine());

            Console.WriteLine("Informe quantidade de horas trabalhadas: ");
            horasTrabalhadas = double.Parse(Console.ReadLine());

            Console.WriteLine("Informe valor que recebe por hora: ");
            valorHora = double.Parse(Console.ReadLine(), CultureInfo.InvariantCulture);

            salario = (valorHora * horasTrabalhadas);

            Console.WriteLine("Number:" + num);
            Console.WriteLine("Salary = U$ " + salario.ToString("F2", CultureInfo.InvariantCulture));


        }
    }
}

5 - Fazer um programa para ler o código de uma peça 1, o número de peças 1, o valor unitário de cada peça 1, o 
código de uma peça 2, o número de peças 2 e o valor unitário de cada peça 2. Calcule e mostre o valor a ser pago

using System;
using System.Globalization;

namespace FirtsProject {
    class Program {
        static void Main(string[] args) {

            double valorPedente = 0.0;

            Console.WriteLine("Informe o codigo da peça um, número da peça um e valor (Mesma linha): ");
            string[] vetUm = Console.ReadLine().Split(' ');
            
            int codPecaUm = int.Parse(vetUm[0]);
            int numPecaUm = int.Parse(vetUm[1]);
            double valorPeçaUm = double.Parse(vetUm[2], CultureInfo.InvariantCulture);


            Console.WriteLine("Informe o codigo da peça dois, número da peça dois e valor (Mesma linha): ");
            string[] vetDois = Console.ReadLine().Split(' ');

            int codPecaDois = int.Parse(vetDois[0]);
            int numPecaDois = int.Parse(vetDois[1]);
            double valorPeçaDois = double.Parse(vetDois[2], CultureInfo.InvariantCulture);

            valorPedente  = ((valorPeçaUm * numPecaUm) + (valorPeçaDois * numPecaDois));

            Console.WriteLine("Valor a pagar: R$ " + valorPedente.ToString("F2", CultureInfo.InvariantCulture));

        }
    }
}
